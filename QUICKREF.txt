import numpy as np

# Load model
model = SentenceTransformer('models/loop/final_model')

# Encode texts

text2 = "Another message to compare"

emb2 = model.encode(text2, normalize_embeddings=True)

# Compute similarity
similarity = np.dot(emb1, emb2)  # Cosine similarity

same_author = similarity >= threshold
